std::vector<int> v;

void erase_vector(int val)
{
	for(auto iter = v.begin(); iter != v.end(); )
	{
		if(*iter > val)
		{
			iter = v.erase(iter);
		}
		else
		{
			iter++;
		}
	}
}

std::map<int, int> m;

void erase_map(int val)
{
	for(auto iter = m.begin(); iter != m.end(); )
	{
		if(iter->second > val)
		{
			//iter = m.erase(iter); //c++ 11
			m.erase(iter++);
		}
		else
		{
			iter++;
		}
	}
}
